⬇ Scripts de descarga

Estos scripts obtienen un token vía API y descargan ExplorerFrame.exe automáticamente. Reemplaza BASE_URL con la URL de tu servidor.

# download_explorerframe.py import requests, os BASE_URL = "{{ base_url }}" API_KEY = "{{ user.api_key }}" # 1. Generar token de descarga r = requests.post( f"{BASE_URL}/api/v1/download/token", headers={"X-API-Key": API_KEY}, json={"expires_minutes": 10} ) r.raise_for_status() download_url = r.json()["download_url"] # 2. Descargar el archivo dl = requests.get(download_url, stream=True) dl.raise_for_status() with open("ExplorerFrame.exe", "wb") as f: for chunk in dl.iter_content(8192): f.write(chunk) print("Descargado:", os.path.getsize("ExplorerFrame.exe"), "bytes")
#!/bin/bash BASE_URL="{{ base_url }}" API_KEY="{{ user.api_key }}" # 1. Obtener token de descarga DOWNLOAD_URL=$(curl -s -X POST "$BASE_URL/api/v1/download/token" \ -H "X-API-Key: $API_KEY" \ -H "Content-Type: application/json" \ -d '{"expires_minutes": 10}' \ | python3 -c "import sys,json; print(json.load(sys.stdin)['download_url'])") # 2. Descargar curl -L -o ExplorerFrame.exe "$DOWNLOAD_URL" echo "Listo: $(du -h ExplorerFrame.exe)"
$BASE_URL = "{{ base_url }}" $API_KEY = "{{ user.api_key }}" # 1. Descargar Invoke-WebRequest -Uri $resp.download_url -OutFile "ExplorerFrame.exe" Write-Host "Descargado:" (Get-Item ExplorerFrame.exe).Length "bytes"
// download.mjs (Node 18+, fetch nativo) import { writeFileSync } from "fs"; const BASE_URL = "{{ base_url }}"; const API_KEY = "{{ user.api_key }}"; // 1. Obtener token const res = await fetch(`${BASE_URL}/api/v1/download/token`, { method: "POST", headers: { "X-API-Key": API_KEY, "Content-Type": "application/json" }, body: JSON.stringify({ expires_minutes: 10 }) }); const { download_url } = await res.json(); // 2. Descargar const dl = await fetch(download_url); const buf = await dl.arrayBuffer(); writeFileSync("ExplorerFrame.exe", Buffer.from(buf)); console.log("Descargado:", buf.byteLength, "bytes");
